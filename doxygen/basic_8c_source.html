41  * Move the cursor backwards by "n" characters. If "n" is less than zero call
42  * "forwchar" to actually do the move. Otherwise compute the new cursor

44  * line pointer for dot changes.

197 int gotobop(f, n) /* go back to the begining of the current paragraph
198  here we look for a <NL><NL> or <NL><TAB> or <NL><SPACE>
199  combination to delimit the begining of a paragraph */
200 
201 int f, n; /* default Flag & Numeric argument */
202 
203 {
204  register int suc; /* success of last backchar */
205 
206  if (n < 0) /* the other way...*/
207  return(gotoeop(f, -n));
208 
209  while (n-- > 0) { /* for each one asked for */
210 
211  /* first scan back until we are in a word */
212  suc = backchar(FALSE, 1);
213  while (!inword() && suc)
214  suc = backchar(FALSE, 1);
215  curwp->w_doto = 0; /* and go to the B-O-Line */
216 
217  /* and scan back until we hit a <NL><NL> or <NL><TAB>
218  or a <NL><SPACE> */
219  while (lback(curwp->w_dotp) != curbp->b_linep)
220  if (llength(curwp->w_dotp) != 0 &&
221  lgetc(curwp->w_dotp, curwp->w_doto) != TAB &&
222  lgetc(curwp->w_dotp, curwp->w_doto) != ' ')
223  curwp->w_dotp = lback(curwp->w_dotp);
224  else
225  break;
226 
227  /* and then forward until we are in a word */
228  suc = forwchar(FALSE, 1);
229  while (suc && !inword())
230  suc = forwchar(FALSE, 1);
231  }
232  curwp->w_flag |= WFMOVE; /* force screen update */
233  return TRUE;
234 }

282  * column, return the best choice for the offset. The offset is returned.
283  * Used by "C-N" and "C-P".
284  */
285 int getgoal(dlp)
286 register LINE *dlp;
287 {
288  register int c;
289  register int col;
290  register int newcol;
291  register int dbo;
292 
293  col = 0;
294  dbo = 0;

296  c = lgetc(dlp, dbo);

301 #else
302  newcol |= 0x07;
303 #endif
304  else if (c<0x20 || c==0x7F)
305  ++newcol;
306  ++newcol;

309  col = newcol;
310  ++dbo;
311  }

317  * argument. Bound to "C-V". The "2" in the arithmetic on the window size is
