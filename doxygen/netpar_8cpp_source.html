802  int i, j, k;
803  PreSyn* ps;
804 
805  // how many gids are there on this machine
806  // and can they be compressed into one byte
807  int ngid = 0;

832  sbuf[ngid] = ps->output_index_;
833  ++ngid;
834  }
835  }}}
