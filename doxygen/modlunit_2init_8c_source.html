170 void init()
171 {
172  int i;
173  Symbol *s;
174 
175  symbol_init();
176  for (i = 0; keywords[i].name; i++) {
177  s = install(keywords[i].name, keywords[i].kval);
178  s->subtype = KEYWORD;
179  }
180  for (i = 0; methods[i].name; i++) {
181  s = install(methods[i].name, METHOD);
182  s->subtype = methods[i].subtype;
183  s->u.i = methods[i].varstep;
184  }
185  for (i = 0; special[i].name; i++) {
186  s = install(special[i].name, SPECIAL);
187  *(special[i].p) = s;
188  s->subtype = special[i].subtype;
189  }
190  for (i = 0; extdef[i]; i++) {
191  s = install(extdef[i], NAME);
192  s->subtype = EXTDEF;
193  }
194  for (i = 0; extargs[i]; ++i) {
195  List* lu;
196  s = lookup(extargs[i++]);
197  assert(s);
198  s->u.str = extargs[i++];
199  lu = newlist();
200  while (extargs[i]) {
201  lappendstr(lu, extargs[i]);
202  ++i;
203  }
204  s->info = itemarray(3,ITEM0, ITEM0, lu);
205  }
