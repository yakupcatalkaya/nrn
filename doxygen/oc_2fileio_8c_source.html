144 const char* expand_env_var(const char* s) {
145  static HocStr* hs;
146  const char* cp1;
147  char* cp2;
148  int n;
149  int begin = 1; /* only needed for mac when necessary to prepend a : */
150  if (!hs) { hs = hocstr_create(256); }
151  hocstr_resize(hs, strlen(s) + 2);
152  for (cp1=s, cp2 = hs->buf + begin; *cp1; ++cp1) {
153  if (*cp1 == '$' && cp1[1] == '(') {
154  char* cp3;

156  cp1 += 2;
157  for (cp3 = buf; *cp1 && *cp1 != ')'; ++cp1) {
158  *cp3++ = *cp1;

173  *cp2++ = *cp3++;
174  }
175  }
176  }else{
177  --cp1;
178  }
179  }else{
180  *cp2++ = *cp1;
181  }
182  }
183  *cp2 = '\0';
184 #if MAC && !defined(DARWIN)
185 /* convert / to : */
186  for (cp1 = hs->buf+begin; *cp1; ++cp1) {
187  if (*cp1 == '/') {
188  *cp1 = ':';
189  }
190  }
191 /* if a : in the original name then assume already mac and done */
192 /* if $ in original name then done since NEURONHOME already has correct prefix */
193 /* if first is : then remove it, otherwise prepend it */
194  if (!strchr(s, ':') && !strchr(s, '$')) {

197  }else{
198  begin = 0;
199  hs->buf[0] = ':';
200  }
201  }

211  }

664  and search in the current, $HOC_LIBRARY_PATH,
665  $NEURONHOME/lib/hoc directories (in that order) for

704  strncpy(expname, expand_env_var(name), hoc_load_file_size_);
705  assert(expname[hoc_load_file_size_ - 1] == '\0');
706  name = expname;

708  strncpy(path, name, base-name);

713 #else
714  f = fopen(name, "r");

716  }else{

720 #if USE_NRNFILEWRAP
721  f = nrn_fw_readaccess(base);
