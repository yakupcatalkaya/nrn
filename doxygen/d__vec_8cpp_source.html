115 appropriate for most usage but not when vectors are used to represent streams

314 static inline void SWAP(double* A, double* B)
315 {
316  double tmp = *A; *A = *B; *B = tmp;

319 /* This should be replaced by a standard ANSI macro. */
320 #define BYTES_PER_WORD 8
321 #define BYTES_PER_LONG 4
322 
323 /* The next 4 #defines implement a very fast in-line stack abstraction. */
324 
325 #define STACK_SIZE (BYTES_PER_WORD * BYTES_PER_LONG)
326 #define PUSH(LOW,HIGH) do {top->lo = LOW;top++->hi = HIGH;} while (0)
327 #define POP(LOW,HIGH) do {LOW = (--top)->lo;HIGH = top->hi;} while (0)
328 #define STACK_NOT_EMPTY (stack < top)
329 
330 /* Discontinue quicksort algorithm when partition gets below this size.
331  This particular magic number was chosen to work best on a Sun 4/260. */
332 #define MAX_THRESH 4
333 
334 
335 /* Order size using quicksort. This implementation incorporates
336  four optimizations discussed in Sedgewick:
337 
338  1. Non-recursive, using an explicit stack of pointer that
339  store the next array partition to sort. To save time, this
340  maximum amount of space required to store an array of
341  MAX_INT is allocated on the stack. Assuming a 32-bit integer,
342  this needs only 32 * sizeof (stack_node) == 136 bits. Pretty
343  cheap, actually.
344 
345  2. Chose the pivot element using a median-of-three decision tree.
346  This reduces the probability of selecting a bad pivot value and
347  eliminates certain extraneous comparisons.
348 
349  3. Only quicksorts TOTAL_ELEMS / MAX_THRESH partitions, leaving
350  insertion sort to order the MAX_THRESH items within each partition.
351  This is a big win, since insertion sort is faster for small, mostly
352  sorted array segements.
353 
354  4. The larger of the two sub-partitions is always pushed onto the
355  stack first, with the algorithm then concentrating on the
356  smaller partition. This *guarantees* no more than log (n)
357  stack size is needed! */
358 
359 static int gsort (double *base_ptr, int total_elems, doubleComparator cmp)
360 {
361 /* Stack node declarations used to store unfulfilled partition obligations. */
362  struct stack_node { double *lo; double *hi; };
363  double pivot_buffer;
364  int max_thresh = MAX_THRESH;
365 
366  if (total_elems > MAX_THRESH)
367  {
368  double *lo = base_ptr;
369  double *hi = lo + (total_elems - 1);
370  double *left_ptr;
371  double *right_ptr;
372  stack_node stack[STACK_SIZE]; /* Largest size needed for 32-bit int!!! */
373  stack_node *top = stack + 1;
374 
375  while (STACK_NOT_EMPTY)
376  {
377  {
378  double *pivot = &pivot_buffer;
379  {
380  /* Select median value from among LO, MID, and HI. Rearrange
381  LO and HI so the three values are sorted. This lowers the
382  probability of picking a pathological pivot value and
383  skips a comparison for both the LEFT_PTR and RIGHT_PTR. */
384 
385  double *mid = lo + ((hi - lo) >> 1);
386 
387  if ((*cmp) (*mid, *lo) < 0)
388  SWAP (mid, lo);
389  if ((*cmp) (*hi, *mid) < 0)
390  {
391  SWAP (mid, hi);
392  if ((*cmp) (*mid, *lo) < 0)
393  SWAP (mid, lo);
394  }
395  *pivot = *mid;
396  pivot = &pivot_buffer;
397  }
398  left_ptr = lo + 1;
399  right_ptr = hi - 1;
400 
401  /* Here's the famous ``collapse the walls'' section of quicksort.
402  Gotta like those tight inner loops! They are the main reason
403  that this algorithm runs much faster than others. */
404  do
405  {
406  while ((*cmp) (*left_ptr, *pivot) < 0)
407  left_ptr += 1;
408 
409  while ((*cmp) (*pivot, *right_ptr) < 0)
410  right_ptr -= 1;
411 
412  if (left_ptr < right_ptr)
413  {
414  SWAP (left_ptr, right_ptr);
415  left_ptr += 1;
416  right_ptr -= 1;
417  }
418  else if (left_ptr == right_ptr)
419  {
420  left_ptr += 1;
421  right_ptr -= 1;
422  break;
423  }
424  }
425  while (left_ptr <= right_ptr);
426 
427  }
428 
429  /* Set up pointers for next iteration. First determine whether
430  left and right partitions are below the threshold size. If so,
431  ignore one or both. Otherwise, push the larger partition's
432  bounds on the stack and continue sorting the smaller one. */
433 
434  if ((right_ptr - lo) <= max_thresh)
435  {
436  if ((hi - left_ptr) <= max_thresh) /* Ignore both small partitions. */
437  POP (lo, hi);
438  else /* Ignore small left partition. */
439  lo = left_ptr;
440  }
441  else if ((hi - left_ptr) <= max_thresh) /* Ignore small right partition. */
442  hi = right_ptr;
443  else if ((right_ptr - lo) > (hi - left_ptr)) /* Push larger left partition indices. */
444  {
445  PUSH (lo, right_ptr);
446  lo = left_ptr;
447  }
448  else /* Push larger right partition indices. */
449  {
450  PUSH (left_ptr, hi);
451  hi = right_ptr;
452  }
453  }
454  }
455 
456  /* Once the BASE_PTR array is partially sorted by quicksort the rest
457  is completely sorted using insertion sort, since this is efficient
458  for partitions below MAX_THRESH size. BASE_PTR points to the beginning
459  of the array to sort, and END_PTR points at the very last element in
460  the array (*not* one beyond it!). */
461 
462 
463  {
464  double *end_ptr = base_ptr + 1 * (total_elems - 1);
465  double *run_ptr;
466  double *tmp_ptr = base_ptr;
467  double *thresh = (end_ptr < (base_ptr + max_thresh))?
468  end_ptr : (base_ptr + max_thresh);
469 
470  /* Find smallest element in first threshold and place it at the
471  array's beginning. This is the smallest array element,
472  and the operation speeds up insertion sort's inner loop. */
473 
474  for (run_ptr = tmp_ptr + 1; run_ptr <= thresh; run_ptr += 1)
475  if ((*cmp) (*run_ptr, *tmp_ptr) < 0)
476  tmp_ptr = run_ptr;
477 
478  if (tmp_ptr != base_ptr)
479  SWAP (tmp_ptr, base_ptr);
480 
481  /* Insertion sort, running from left-hand-side up to `right-hand-side.'
482  Pretty much straight out of the original GNU qsort routine. */
483 
484  for (run_ptr = base_ptr + 1; (tmp_ptr = run_ptr += 1) <= end_ptr; )
485  {
486 
487  while ((*cmp) (*run_ptr, *(tmp_ptr -= 1)) < 0)
488  ;
489 
490  if ((tmp_ptr += 1) != run_ptr)
491  {
492  double *trav;
493 
494  for (trav = run_ptr + 1; --trav >= run_ptr;)
495  {
496  double c = *trav;
497  double *hi, *lo;
498 
499  for (hi = lo = trav; (lo -= 1) >= tmp_ptr; hi = lo)
500  *hi = *lo;
501  *hi = c;
502  }
503  }
504 
505  }
506  }
507  return 1;
508 }
