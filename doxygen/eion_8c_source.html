484 void nrn_promote(p, conc, rev)
485  Prop* p;
486  int conc, rev;
487 {
488  int oldconc, oldrev;
489  int* it = &p->dparam[0].i;
490  oldconc = (*it & 03);
491  oldrev = (*it & 030)>>3;
492  /* precedence */
493  if (oldconc < conc) {
494  oldconc = conc;
495  }
496  if (oldrev < rev) {
497  oldrev = rev;
498  }
499  /* promote type */
500  if (oldconc > 0 && oldrev < 2) {
501  oldrev = 2;
502  }
503  *it &= ~0177; /* clear the bitmap */
504  *it += oldconc + 010*oldrev;
505  if (oldconc == 3) { /* if state then cinit */
506  *it += 4;
507  if (oldrev == 2) { /* if not state (WRITE) then eadvance */
508  *it += 0100;
509  }
510  }
511  if (oldconc > 0 && oldrev == 2) { /*einit*/
512  *it += 040;
513  }
