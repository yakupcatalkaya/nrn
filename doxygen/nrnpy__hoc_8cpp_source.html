1648 static PyObject* iternext_sl(PyHocObject* po, hoc_Item* ql) {
1649  // Note that the longstanding behavior of changing the currently
1650  // accessed section during iteration no longer takes place because
1651  // we cannot guarantee that an iterate will complete with state
1652  // PyHoc::Last and so the previous Section would have been left on the
1653  // hoc section stack.
1654 
1655  // Primarily the Section is pushed and the currently accessed python
1656  // Section is returned during sequential iteration over the list ql.
1657  // On re-entry here, the previous Section is popped.
1658  // The complexity is due to the possibility that the returned nrn_Section
1659  // may be deleted with h.delete_section(sec=nrn_Section). This would
1660  // invalidate the po->iteritem_ (point to freed memory) if it were
1661  // the iteritem of the current Section. Thus we choose to store
1662  // the next iteritem pointer in the list. Although not 100% safe, since
1663  // the user body of the iterator is allowed to delete_section an arbitary
1664  // subset of sections, the obvious work around, making a copy of ql,
1665  // is considered not worth it.
1666 
1667  // In this implementation, the first call to internext_sl starts out
1668  // in state PyHoc::Begin with po->iteritem_ == ql. If there is no valid
1669  // item, it sets po->iteritem_ = NULL and returns NULL. If there is
1670  // a valid item, it moves to state PyHoc::Last or PyHoc::NextNotLast
1671  // depending on whether there is a valid secitem following the current
1672  // item. The current section is pushed and currently accessed python
1673  // section is returned.
1674 
1675  // Thereafter, re-entry with po->iteritem_ == NULL, immediately
1676  // returns NULL.
1677 
1678  // Re-entry in state PyHoc::NextNotLast, pops the previously pushed Section,
1679  // sets sec to the current Section from po->iteritem_, and
1680  // sets po->iteritem_ to the next_valid_section. If there is no next_valid
1681  // section, then move to state PyHoc::Last.
1682  // Push the current section and return currently accessed python Section.
1683 
1684  // Re-entry in state PyHoc::Last just pops the current section, sets
1685  // po->iteritem_ = NULL, and returns NULL.
1686 
1687  if (po->iteritem_ == NULL) {
1688  return NULL;
1689  }
1690 
1691  if (po->u.its_ == PyHoc::Begin) {
1692  assert(po->iteritem_ == ql);
1693  hoc_Item* curitem = next_valid_secitem((hoc_Item*)(po->iteritem_), ql);
1694  if (curitem != ql) { // typical case, a valid current item
1695  Section* sec = curitem->element.sec;
1696  assert(sec->prop);
1697  // sec could be delete_section before return to internext.sl
1698  // which would invalidate curitem.
1699  // Not perfectly safe but leave po->iteritem_ as next valid
1700  // secitem after curitem.
1701  po->iteritem_ = next_valid_secitem(curitem, ql);
1702  if (po->iteritem_ == ql) {
1703  po->u.its_ = PyHoc::Last;
1704  }else{
1705  po->u.its_ = PyHoc::NextNotLast;
1706  }
1707  return (PyObject*)newpysechelp(sec);
1708  }else{ // no valid current item so stop
1709  po->iteritem_ = NULL;
1710  return NULL;
1711  }
1712  } else if (po->u.its_ == PyHoc::NextNotLast) {
1713  // it would be a bug if po->iteritem_ (now the curitem) has been delete_section
1714  Section* sec = ((hoc_Item*)(po->iteritem_))->element.sec;
1715  if (!sec->prop) {
1716  // Handle an edge case where sec is a Python
1717  // Section and happens to have had as its only reference the
1718  // iteration variable on the previous iteration.
1719  // i.e. when sec is returned from the previous iteration, it will
1720  // be referenced by the iteration variable and the previous reference
1721  // will go to 0, thus invalidating po->iteritem_->element.sec->prop
1722  po->iteritem_ = next_valid_secitem((hoc_Item*)(po->iteritem_), ql);
1723  if (po->iteritem_ == ql) {
1724  po->u.its_ = PyHoc::Last;
1725  po->iteritem_ = NULL;
1726  return NULL;
1727  }else{
1728  sec = ((hoc_Item*)(po->iteritem_))->element.sec;
1729  }
1730  }
1731  assert(sec->prop);
1732  po->iteritem_ = next_valid_secitem((hoc_Item*)(po->iteritem_), ql);
1733  if (po->iteritem_ == ql) {
1734  po->u.its_ = PyHoc::Last;
1735  }
1736  return (PyObject*)newpysechelp(sec);
1737  } else if (po->u.its_ == PyHoc::Last) {
1738  po->iteritem_ = NULL;
1739  return NULL;
1740  }
1741  return NULL; // never get here as po->u.its_ is always a defined state.
1742 }
