227 /*
228  * Split the current window. A window smaller than 3 lines cannot be split.
229  * The only other error that is possible is a "malloc" failure allocating the
230  * structure for the new window. Bound to "C-X 2".
231  */
232 int splitwind(f, n)
233  int f, n;
234 {
235  register WINDOW *wp;
236  register LINE *lp;
237  register int ntru;
238  register int ntrl;
239  register int ntrd;
240  register WINDOW *wp1;
241  register WINDOW *wp2;
242 
243  LINTUSE(f) LINTUSE(n)
244  if (curwp->w_ntrows < 3) {
245  mlwrite("Cannot split a %d line window", curwp->w_ntrows);
246  return (FALSE);
247  }
248  if ((wp = (WINDOW *) malloc(sizeof(WINDOW))) == NULL) {
249  mlwrite("Cannot allocate WINDOW block");
250  return (FALSE);
251  }
252  ++curbp->b_nwnd; /* Displayed twice. */
253  wp->w_bufp = curbp;
254  wp->w_dotp = curwp->w_dotp;
255  wp->w_doto = curwp->w_doto;
256  wp->w_markp = curwp->w_markp;
257  wp->w_marko = curwp->w_marko;
258  wp->w_flag = 0;
259  wp->w_force = 0;
260  ntru = (curwp->w_ntrows-1) / 2; /* Upper size */
261  ntrl = (curwp->w_ntrows-1) - ntru; /* Lower size */
262  lp = curwp->w_linep;
263  ntrd = 0;
264  while (lp != curwp->w_dotp) {
265  ++ntrd;
266  lp = lforw(lp);
267  }
268  lp = curwp->w_linep;
269  if (ntrd <= ntru) { /* Old is upper window. */
270  if (ntrd == ntru) /* Hit mode line. */
271  lp = lforw(lp);
272  curwp->w_ntrows = ntru;
273  wp->w_wndp = curwp->w_wndp;
274  curwp->w_wndp = wp;
275  wp->w_toprow = curwp->w_toprow+ntru+1;
276  wp->w_ntrows = ntrl;
277  } else { /* Old is lower window */
278  wp1 = NULL;
279  wp2 = wheadp;
280  while (wp2 != curwp) {
281  wp1 = wp2;
282  wp2 = wp2->w_wndp;
283  }
284  if (wp1 == NULL)
285  wheadp = wp;
286  else
287  wp1->w_wndp = wp;
288  wp->w_wndp = curwp;
289  wp->w_toprow = curwp->w_toprow;
290  wp->w_ntrows = ntru;
291  ++ntru; /* Mode line. */
292  curwp->w_toprow += ntru;
293  curwp->w_ntrows = ntrl;
294  while (ntru--)
295  lp = lforw(lp);
296  }
297  curwp->w_linep = lp; /* Adjust the top lines */
298  wp->w_linep = lp; /* if necessary. */
299  curwp->w_flag |= WFMODE|WFHARD;
300  wp->w_flag |= WFMODE|WFHARD;
301  return (TRUE);
302 }

306  * is big enough. If so, hack the window descriptions, and ask redisplay to do
307  * all the hard work. You don't just set "force reframe" because dot would
308  * move. Bound to "C-X Z".

310 int enlargewind(f, n)
311  int f, n;
312 {
313  register WINDOW *adjwp;
314  register LINE *lp;
315  register int i;
316 
317  if (n < 0)
318  return (shrinkwind(f, -n));

320  mlwrite("Only one window");
321  return (FALSE);
322  }
323  if ((adjwp=curwp->w_wndp) == NULL) {
324  adjwp = wheadp;
325  while (adjwp->w_wndp != curwp)
326  adjwp = adjwp->w_wndp;
