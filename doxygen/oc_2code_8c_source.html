144 /* temporary object references come from this pool. This allows the

146 solve problems of objrefs on the stack changing the object they point
147 to and also a failure of garbage collection since temporary objrefs have
148 not, in the past, been reffed or unreffed.
149 The first problem is easily solved without much efficiency loss
150 by having the stack store the object pointer instead of the objref pointer.

153 on the stack via hoc_push_object (and thus borrows the use of the

287 static void frame_objauto_recover_on_err(Frame* ff) { /* only on error */
288  Frame* f;
289  for (f = fp; f > ff; --f) {
290  int i;
291  Symbol* sp = f->sp;
292  if (sp->u.u_proc == NULL) { /* skip if the procedure is not defined */
293  continue;
294  }
295  /* argn is the nargs argument on the stack. Stack items come in pairs
296  so stack increments are always multiples of 2.
297  Here, stkp is the last+1 localobj slot pair on the stack.
298  */
299  Datum* stkp = f->argn + 2 + sp->u.u_proc->nauto*2;
300  for (i = sp->u.u_proc->nobjauto; i > 0; --i) {
301  Object* ob = stkp[-2*i].obj;
302  hoc_obj_unref(ob);
303  /* Note that these AUTOOBJECT stack locations have an itemtype that
304  are left over from the previous stack usage of that location.
305  Regardless of that itemtype (e.g. OBJECTTMP), these did NOT
306  increment tobj_count so we need to guarantee that the subsequent
307  stack_obtmp_recover_on_err does not inadvertently free it again
308  by setting the itemtype to a non OBJECTTMP value. I hope this is
309  the only place where stack space was used in which no item type
310  was specified.
311  We are doing this here which happens rarely to avoid having to
312  set them when the stack obj pointers are zeroed.
313  */
314  stkp[-2*i + 1].i = 0;
315  }
316  }
