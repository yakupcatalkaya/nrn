93 /* Return a new keymap with the printing characters bound to rl_insert,
94  the uppercase Meta characters bound to run their lowercase equivalents,
95  and the Meta digits bound to produce numeric arguments. */
96 Keymap
97 rl_make_keymap ()
98 {
99  extern rl_insert (), rl_rubout ();
100  register int i;
101  Keymap newmap;
102 
103  newmap = rl_make_bare_keymap ();
104 
105  /* All printing characters are self-inserting. */
106  for (i = ' '; i < 126; i++)
107  newmap[i].function = rl_insert;
108 
109  newmap[TAB].function = rl_insert;
110  newmap[RUBOUT].function = rl_rubout;
111  newmap[CTRL('H')].function = rl_rubout;
112 
113  return (newmap);
114 }
