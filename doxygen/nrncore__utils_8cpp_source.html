127 // This function is related to stdindex2ptr in CoreNeuron to determine which values should
128 // be transferred from CoreNeuron. Types correspond to the value to be transferred based on
129 // mech_type enum or non-artificial cell mechanisms.
130 // Limited to pointers to voltage, nt._nrn_fast_imem->_nrn_sav_rhs (fast_imem value) or
131 // data of non-artificial cell mechanisms.
132 // Requires cache_efficient mode.
133 // Input double* and NrnThread. Output type and index.
134 // type == 0 means could not determine index.
135 int nrn_dblpntr2nrncore(double* pd, NrnThread& nt, int& type, int& index) {
136  assert(use_cachevec);
137  int nnode = nt.end;
138  type = 0;
139  if (pd >= nt._actual_v && pd < (nt._actual_v + nnode)) {
140  type = voltage; // signifies an index into voltage array portion of _data
141  index = pd - nt._actual_v;
142  } else if (nt._nrn_fast_imem && pd >= nt._nrn_fast_imem->_nrn_sav_rhs && pd < (nt._nrn_fast_imem->_nrn_sav_rhs + nnode)) {
143  type = i_membrane_; // signifies an index into i_membrane_ array portion of _data
144  index = pd - nt._nrn_fast_imem->_nrn_sav_rhs;
145  }else{
146  for (NrnThreadMembList* tml = nt.tml; tml; tml = tml->next) {
147  if (nrn_is_artificial_[tml->index]) { continue; }
148  Memb_list* ml1 = tml->ml;
149  int nn = nrn_prop_param_size_[tml->index] * ml1->nodecount;
150  if (pd >= ml1->data[0] && pd < (ml1->data[0] + nn)) {
151  type = tml->index;
152  index = pd - ml1->data[0];
153  break;
154  }
155  }
156  }
157  return type == 0 ? 1 : 0;
158 }
