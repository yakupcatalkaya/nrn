105 uint32_t nrnRan4int(uint32_t* idx1, uint32_t idx2)
106 {
107  uint32_t u, v, w, m, n;
108  /* 64-bit hash */
109  n = (*idx1)++;
110  m = idx2;
111 
112  w = n ^ 0xbaa96887;
113  v = w >> 16;
114  w &= 0xffff;
115  u = ~((v - w) * (v + w));
116  /*m ^= (((u >> 16) | (u << 16)) ^ 0xb4f0c4a7) + w * v;*/
117  m ^= (((u >> 16) | (u << 16)) ^ 0x4b0f3b58) + w * v;
118 
119  w = m ^ 0x1e17d32c;
120  v = w >> 16;
121  w &= 0xffff;
122  u = ~((v - w) * (v + w));
123  /*n ^= (((u >> 16) | (u << 16)) ^ 0x178b0f3c) + w * v;*/
124  n ^= (((u >> 16) | (u << 16)) ^ 0xe874f0c3) + w * v;
125  return n;
126 
127  w = n ^ 0x03bcdc3c;
128  v = w >> 16;
129  w &= 0xffff;
130  u = (v - w) * (v + w);
131  m ^= (((u >> 16) | (u << 16)) ^ 0x96aa3a59) + w * v;
132

134  v = w >> 16;

141 /*
142 //+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
143 // Function: Randouble (hines now nrnRan4dbl
144 //
145 // Purpose: Random double-precision floating-point sample.
146 // The 2^52 allowed values are odd multiples of 2^-53,
147 // symmetrically placed in strict interior of (0,1).
148 //
149 // Notes: (1) Tuned to 52-bit mantissa in "double" format.
150 // (2) Uses one call to Ranint to get 64 random bits, with extra
151 // random integer available in Rand[3].
152 // (3) All floating-point random calls are directed through this code,
153 // except Rangauss which uses the extra random integer in Rand[3].
154 //
155 // History: John Skilling 6 May 1995, 3 Dec 1995, 24 Aug 1996
156 // 20 Oct 2002, 17 Dec 2002
157 //-----------------------------------------------------------------------------
158 //
159 */
160 static const double SHIFT32 = 1.0 / 4294967296.0; /* 2^-32 */
161 double nrnRan4dbl(uint32_t* idx1, uint32_t idx2)
162 {
163  uint32_t hi, lo, extra;
164  hi = (uint32_t)nrnRan4int(idx1, idx2); /*top 32 bits*/
165 /*
166 // lo = (extra // low bits
167 // & 0xfffff000) ^ 0x00000800; // switch lowest (2^-53) bit ON
168 // return ((double)hi + (double)lo * SHIFT32) * SHIFT32;
169 */
170  return ((double)hi) * SHIFT32;
171 }
