1358 static PyObject* seg_volume(NPySegObj* self) {
1359  Section* sec = self->pysec_->sec_;
1360  int i;
1361  if (sec->recalc_area_) {
1362  nrn_area_ri(sec);
1363  }
1364  double x = self->x_;
1365  double a = 0.0;
1366  // volume only exists on the interior
1367  if (x > 0. && x < 1.) {
1368  // every section owns one extra node (the conservation node)
1369  int nseg = sec->nnode - 1;
1370  double length = section_length(sec) / nseg;
1371  int iseg = x * nseg;
1372  double seg_left_arc = iseg * length;
1373  double seg_right_arc = (iseg + 1) * length;
1374  if (sec->npt3d > 1) {
1375  int i_left = arg_bisect_arc3d(sec, sec->npt3d, seg_left_arc);
1376  double left_diam = fabs(sec->pt3d[i_left].d);
1377  double right_diam = fabs(sec->pt3d[i_left + 1].d);
1378  double left_arc = seg_left_arc;
1379  left_diam = interpolate(sec->pt3d[i_left].arc, sec->pt3d[i_left + 1].arc, left_diam, right_diam, left_arc);
1380 
1381  for (i = i_left + 1; i < sec->npt3d && sec->pt3d[i].arc < seg_right_arc; i++) {
1382  right_diam = fabs(sec->pt3d[i].d);
1383  a += scaled_frustum_volume(sec->pt3d[i].arc - left_arc, left_diam, right_diam);
1384  left_arc = sec->pt3d[i].arc;
1385  left_diam = right_diam;
1386  }
1387  if (i < sec->npt3d) {
1388  right_diam = interpolate(left_arc, sec->pt3d[i].arc, left_diam, fabs(sec->pt3d[i].d), seg_right_arc);
1389  a += scaled_frustum_volume(seg_right_arc - left_arc, left_diam, right_diam);
1390  }
1391 
1392  // the scaled_frustum_volume formula factored out a pi/12
1393  a *= M_PI / 12;
1394  } else {
1395  // 0 or 1 3D points... so give cylinder volume
1396  Node* nd = node_exact(sec, x);
1397  for (Prop* p = nd->prop; p; p = p->next) {
1398  if (p->type == MORPHOLOGY) {
1399  double diam = p->param[0];
1400  a = M_PI * diam * diam / 4 * length;
1401  break;
1402  }
1403  }
1404  }
1405  }
1406  PyObject* result = Py_BuildValue("d", a);
1407  return result;
1408 }
