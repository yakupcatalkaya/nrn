342  const Font* f = kit.font();
343  FontBoundingBox bbox;
344  f->font_bbox(bbox);
345  Coord height = rows * (bbox.ascent() + bbox.descent()) + 1.0;
346  Coord width;

348  width = 16 * f->width('m') + 3.0;

358  editor_ = DialogKit::instance()->field_editor(
359  "", s,

361  this, &SymChooserImpl::editor_accept, NULL
362  )
363  );
364  browser_index_ = 0;
365  for (i=0; i < nbrowser_; ++i) {
366  fbrowser_[i] = new FileBrowser(kit_,
367  new SymBrowserAccept(this, i),

369  }
370  fchooser_->remove_all_input_handlers();
371  fchooser_->append_input_handler(editor_);
372  for (i=0; i < nbrowser_; ++i) {
373  fchooser_->append_input_handler(fbrowser_[i]);

376

386  g->append(layout.vglue(5.0, 0.0, 2.0));

388  g->append(layout.vglue(15.0, 0.0, 12.0));

390  Glyph* b;

392  b = layout.hbox(
393  layout.vcenter(

538 void SymChooserImpl::load(int bindex) {
539  SymDirectory& d = *dir_[bindex];
540  Browser& b = *fbrowser_[bindex];
541  WidgetKit& kit = *kit_;
542  kit.push_style();
543  kit.style(style_);
544  const LayoutKit& layout = *LayoutKit::instance();
545  int dircount = d.count();
546  delete [] filter_map_;
547  int* index = new int[dircount];
548  filter_map_ = index;
549 //printf("loading %d\n", bindex);
550  for (int i = 0; i < dircount; i++) {
551  const String& f = d.name(i);
552  bool is_dir = d.is_directory(i);
553  if ((is_dir && filtered(f, directory_filter_)) ||
554  (!is_dir && filtered(f, filter_))
555  ) {
556  Glyph* name = kit.label(f);
557  if (is_dir) {
558  if (d.symbol(i) && d.symbol(i)->type == TEMPLATE) {
559  name = layout.hbox(name, kit.label("_"));
560  }else{
561  name = layout.hbox(name, kit.label("."));
562  }
563  }
564  Glyph* label = new Target(
565  layout.h_margin(name, 3.0, 0.0, 0.0, 15.0, fil, 0.0),
566  TargetPrimitiveHit
567  );
568  TelltaleState* t = new TelltaleState(TelltaleState::is_enabled);
569  b.append_selectable(t);
570  b.append(new ChoiceItem(t, label, kit.bright_inset_frame(label)));
571  *index++ = i;
572  }
573  }
574  // the order of the following two statements is important for carbon
575  // to avoid a premature browser request which ends up showing an
576  // empty list.
577  fbrowser_[bindex]->refresh();
578  editor_->field(d.path());
579  kit.pop_style();
580 }

618 void SymChooserImpl::accept_browser_index(int bindex) {
619  int i = int(fbrowser_[bindex]->selected());
620  if (i == -1) {
621  return;
622  }
623 // i = filter_map_[i];
624  SymDirectory* dir = dir_[bindex];
625  const String& path = dir->path();
626  const String& name = dir->name(i);
627  Symbol* sym = dir->symbol(i);
628  int length = path.length() + name.length();
629  char* tmp = new char[length + 2];
630  sprintf(

636  last_index_ = i;

642  /* should generate an error message */
643  }
644  } else {
645  clear(bindex+1);
646  browser_index_ = bindex;
647 // fchooser_->dismiss(true);
648  }
649  delete [] tmp;
650 }
