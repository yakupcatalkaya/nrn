540  /*
541  treesolver
542 
543  a - above the diagonal
544  b - below the diagonal
545  c - used to define diagonal: diag = c + dt * dbase
546  dbase - together with c, used to define diagonal
547  rhs - right hand side, which is changed to the result
548  pindex - parent indices
549  n - number of states
550  */
551  long i, pin;
552  double* d = (double*)malloc(sizeof(double) * n);
553  double* myd;
554  double* myc;
555  double* mydbase;
556 
557  /* TODO: check that d non-null */
558 
559  /* construct diagonal */
560  myd = d;
561  mydbase = dbase;
562  myc = c;
563  for (i = 0; i < n; i++) {
564  *myd++ = *myc++ + dt * (*mydbase++);
565  }
566 
567  /* triang */
568  for (i = n - 1; i > 0; --i) {
569  pin = pindex[i];
570  if (pin > -1) {
571  double p;
572  p = dt * a[i] / d[i];
573  d[pin] -= dt * p * b[i];
574  rhs[pin] -= p * rhs[i];
575  }
576  }
577  /* bksub */
578  for (i = 0; i < n; ++i) {
579  pin = pindex[i];
580  if (pin > -1) {
581  rhs[i] -= dt * b[i] * rhs[pin];
582  }
583  rhs[i] /= d[i];
584  }
585 
586  free(d);
587 }
